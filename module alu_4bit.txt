module alu_4bit (
    input  logic [3:0] A, B,
    input  logic [2:0] sel,
    output logic [3:0] Y
);
    always_comb begin
        case(sel)
            3'b000: Y = A + B;       
            3'b001: Y = A - B;       
            3'b010: Y = A & B;      
            3'b011: Y = A | B;       
            3'b100: Y = A ^ B;       
            3'b101: Y = A << 1;      
            3'b110: Y = A >> 1;      
            default: Y = 4'b0000;    
        endcase
    end
endmodule
